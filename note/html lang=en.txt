<script>
    document.addEventListener('DOMContentLoaded', () => {
        // Retrieve all message elements
        const messages = Array.from(document.querySelectorAll('.message'));
        
        // background colors 
        const messageColors = {
            "message1": "#FJA552",
            
        };

        let currentIdx = 0; // Global index of the currently shown message

        // Add smooth transition to body
        document.body.style.transition = 'background-color 0.8s ease';

        // --- Message Display Logic ---
        function showMessage(idx) {
            // Ensure index is valid
            if (idx < 0 || idx >= messages.length) return;

            messages.forEach(m => m.classList.remove('visible'));
            messages[idx].classList.add('visible');
            
            // Update global index
            currentIdx = idx; 
            
            // Change background color based on message ID with smooth transition
            const messageId = messages[idx].id;
            const bgColor = messageColors[messageId] || '#f8f8f8';
            document.body.style.backgroundColor = bgColor;
        }

        // --- Advance Message Logic ---
        function advanceMessage() {
            let nextIdx = currentIdx + 1;
            
            if (nextIdx >= messages.length) {
                // Loop back to first message
                nextIdx = 0;
            }
            
            showMessage(nextIdx);
        }
        
        // Add click event to advance message when clicking anywhere on the page
        document.body.addEventListener('click', advanceMessage);

        // Prevent music player clicks from advancing slides
        const musicPlayer = document.getElementById('musicPlayer');
        if (musicPlayer) {
            musicPlayer.addEventListener('click', (e) => {
                e.stopPropagation();
            });
        }

        // Show first message
        showMessage(0);
    });
</script>
